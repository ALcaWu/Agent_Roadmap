def reverse_string(s):

def is_even(num):
    return num % 2 == 0

# 数字统计（统计1-100中能被3整除的数）
def count_divisible_by_3():
    count = 0
    for i in range(1, 101):
        if i % 3 == 0:
            count += 1
    return count
